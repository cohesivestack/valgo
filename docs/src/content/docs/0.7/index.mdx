Valgo is a type-safe, expressive, and extensible validator library for Go. Rules are written in functions (not struct tags), which makes validation logic easier to compose, reuse, and test.

```bash
go get github.com/cohesivestack/valgo
```

```go
package main

import (
  "encoding/json"
  "fmt"

  v "github.com/cohesivestack/valgo"
)

func main() {
  val := v.Is(
    v.String("Bob", "full_name").Not().Blank().OfLengthBetween(4, 20),
    v.Number(17, "age").GreaterThan(18),
  )

  if !val.Valid() {
    out, _ := json.MarshalIndent(val.ToError(), "", "  ")
    fmt.Println(string(out))
  }
}
```
